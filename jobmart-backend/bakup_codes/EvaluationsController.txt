using System.Security.Claims;
using jobmart.Data;
using jobmart.DTOs;
using jobmart.Models;
using Microsoft.AspNetCore.Authorization;
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;

namespace jobmart.bakup_codes
{
    [Route("api/[controller]")]
    [ApiController]
    [Authorize(Roles = "HiringManager,Admin")]
    public class EvaluationsController : ControllerBase
    {
        private readonly ApplicationDbContext _context;

        public EvaluationsController(
            ApplicationDbContext context)
        {
            _context = context;
        }

        // --------------------------------------------------
        // CREATE CANDIDATE EVALUATION
        // POST: /api/Evaluations
        // --------------------------------------------------
        [HttpPost]
        public async Task<IActionResult> CreateEvaluation(
            CreateEvaluationDto request)
        {
            int? currentUserId = GetCurrentUserId();

            if (currentUserId == null)
            {
                return Unauthorized(new
                {
                    message = "Invalid user token."
                });
            }

            var application = await _context.JobApplications
                .Include(application => application.JobPosting)
                .Include(application => application.CandidateProfile)
                    .ThenInclude(profile => profile!.User)
                .FirstOrDefaultAsync(application =>
                    application.JobApplicationId ==
                    request.JobApplicationId
                );

            if (application == null)
            {
                return NotFound(new
                {
                    message = "Job application not found."
                });
            }

            var allowedApplicationStatuses = new[]
            {
                "Shortlisted",
                "InterviewScheduled",
                "UnderReview"
            };

            if (!allowedApplicationStatuses.Contains(
                    application.Status))
            {
                return BadRequest(new
                {
                    message =
                        "Only shortlisted, reviewed, or interviewed candidates can be evaluated."
                });
            }

            var allowedRecommendations = new[]
            {
                "RecommendHire",
                "Consider",
                "Reject"
            };

            var normalizedRecommendation =
                allowedRecommendations.FirstOrDefault(
                    recommendation =>
                        recommendation.Equals(
                            request.Recommendation,
                            StringComparison.OrdinalIgnoreCase
                        )
                );

            if (normalizedRecommendation == null)
            {
                return BadRequest(new
                {
                    message = "Invalid recommendation.",
                    allowedRecommendations
                });
            }

            bool evaluationExists =
                await _context.CandidateEvaluations
                    .AnyAsync(evaluation =>
                        evaluation.JobApplicationId ==
                            request.JobApplicationId &&
                        evaluation.HiringManagerId ==
                            currentUserId.Value
                    );

            if (evaluationExists)
            {
                return BadRequest(new
                {
                    message =
                        "You have already evaluated this application."
                });
            }

            decimal overallScore =
                CalculateOverallScore(request);

            var evaluation = new CandidateEvaluation
            {
                JobApplicationId =
                    request.JobApplicationId,

                HiringManagerId =
                    currentUserId.Value,

                TechnicalScore =
                    request.TechnicalScore,

                CommunicationScore =
                    request.CommunicationScore,

                ExperienceScore =
                    request.ExperienceScore,

                ProblemSolvingScore =
                    request.ProblemSolvingScore,

                CultureFitScore =
                    request.CultureFitScore,

                OverallScore = overallScore,

                Feedback = request.Feedback?.Trim(),

                Recommendation =
                    normalizedRecommendation,

                Decision = "Pending",

                CreatedAt = DateTime.UtcNow,

                UpdatedAt = DateTime.UtcNow
            };

            _context.CandidateEvaluations.Add(evaluation);

            await _context.SaveChangesAsync();

            return Ok(new
            {
                message =
                    "Candidate evaluation created successfully.",

                evaluation = new
                {
                    evaluation.EvaluationId,
                    evaluation.JobApplicationId,
                    evaluation.HiringManagerId,
                    evaluation.TechnicalScore,
                    evaluation.CommunicationScore,
                    evaluation.ExperienceScore,
                    evaluation.ProblemSolvingScore,
                    evaluation.CultureFitScore,
                    evaluation.OverallScore,
                    evaluation.Feedback,
                    evaluation.Recommendation,
                    evaluation.Decision,
                    evaluation.CreatedAt
                }
            });
        }

        // --------------------------------------------------
        // VIEW EVALUATIONS FOR AN APPLICATION
        // GET:
        // /api/Evaluations/application/1
        // --------------------------------------------------
        [HttpGet("application/{applicationId}")]
        public async Task<IActionResult>
            GetEvaluationsForApplication(
                int applicationId)
        {
            var applicationExists =
                await _context.JobApplications
                    .AnyAsync(application =>
                        application.JobApplicationId ==
                        applicationId
                    );

            if (!applicationExists)
            {
                return NotFound(new
                {
                    message = "Job application not found."
                });
            }

            var evaluations =
                await _context.CandidateEvaluations
                    .Include(evaluation =>
                        evaluation.HiringManager)
                    .Where(evaluation =>
                        evaluation.JobApplicationId ==
                        applicationId
                    )
                    .OrderByDescending(evaluation =>
                        evaluation.CreatedAt)
                    .Select(evaluation => new
                    {
                        evaluation.EvaluationId,
                        evaluation.JobApplicationId,
                        evaluation.HiringManagerId,

                        HiringManagerName =
                            evaluation.HiringManager != null
                                ? evaluation.HiringManager
                                    .FullName
                                : null,

                        HiringManagerEmail =
                            evaluation.HiringManager != null
                                ? evaluation.HiringManager
                                    .Email
                                : null,

                        evaluation.TechnicalScore,
                        evaluation.CommunicationScore,
                        evaluation.ExperienceScore,
                        evaluation.ProblemSolvingScore,
                        evaluation.CultureFitScore,
                        evaluation.OverallScore,
                        evaluation.Feedback,
                        evaluation.Recommendation,
                        evaluation.Decision,
                        evaluation.CreatedAt,
                        evaluation.UpdatedAt
                    })
                    .ToListAsync();

            return Ok(evaluations);
        }

        // --------------------------------------------------
        // UPDATE HIRING DECISION
        // PUT:
        // /api/Evaluations/application/1/decision
        // --------------------------------------------------
        [HttpPut("application/{applicationId}/decision")]
        public async Task<IActionResult>
            UpdateHiringDecision(
                int applicationId,
                HiringDecisionDto request)
        {
            int? currentUserId = GetCurrentUserId();

            if (currentUserId == null)
            {
                return Unauthorized(new
                {
                    message = "Invalid user token."
                });
            }

            var allowedDecisions = new[]
            {
                "Hired",
                "Rejected",
                "Pending"
            };

            var normalizedDecision =
                allowedDecisions.FirstOrDefault(
                    decision =>
                        decision.Equals(
                            request.Decision,
                            StringComparison.OrdinalIgnoreCase
                        )
                );

            if (normalizedDecision == null)
            {
                return BadRequest(new
                {
                    message = "Invalid hiring decision.",
                    allowedDecisions
                });
            }

            var application = await _context.JobApplications
                .Include(application => application.JobPosting)
                .Include(application =>
                    application.CandidateProfile)
                    .ThenInclude(profile => profile!.User)
                .FirstOrDefaultAsync(application =>
                    application.JobApplicationId ==
                    applicationId
                );

            if (application == null)
            {
                return NotFound(new
                {
                    message = "Job application not found."
                });
            }

            var evaluation =
                await _context.CandidateEvaluations
                    .FirstOrDefaultAsync(evaluation =>
                        evaluation.JobApplicationId ==
                            applicationId &&
                        evaluation.HiringManagerId ==
                            currentUserId.Value
                    );

            if (evaluation == null)
            {
                return BadRequest(new
                {
                    message =
                        "You must create an evaluation before making a hiring decision."
                });
            }

            evaluation.Decision =
                normalizedDecision;

            evaluation.UpdatedAt =
                DateTime.UtcNow;

            if (normalizedDecision == "Hired")
            {
                application.Status = "Hired";
            }
            else if (normalizedDecision == "Rejected")
            {
                application.Status = "Rejected";
            }
            else
            {
                application.Status = "UnderReview";
            }

            await _context.SaveChangesAsync();

            return Ok(new
            {
                message =
                    "Hiring decision updated successfully.",

                applicationId =
                    application.JobApplicationId,

                applicationStatus =
                    application.Status,

                evaluationId =
                    evaluation.EvaluationId,

                decision =
                    evaluation.Decision
            });
        }

        // --------------------------------------------------
        // VIEW CANDIDATE DETAILS FOR REVIEW
        // GET:
        // /api/Evaluations/candidate/application/1
        // --------------------------------------------------
        [HttpGet("candidate/application/{applicationId}")]
        public async Task<IActionResult>
            GetCandidateForEvaluation(
                int applicationId)
        {
            var application =
                await _context.JobApplications
                    .Include(application =>
                        application.JobPosting)
                    .Include(application =>
                        application.CandidateProfile)
                        .ThenInclude(profile =>
                            profile!.User)
                    .Include(application =>
                        application.CandidateProfile)
                    .FirstOrDefaultAsync(application =>
                        application.JobApplicationId ==
                        applicationId
                    );

            if (application == null)
            {
                return NotFound(new
                {
                    message = "Job application not found."
                });
            }

            return Ok(new
            {
                application.JobApplicationId,
                application.Status,
                application.AIScore,
                application.AppliedAt,

                job = new
                {
                    application.JobId,

                    title =
                        application.JobPosting?.Title,

                    location =
                        application.JobPosting?.Location,

                    employmentType =
                        application.JobPosting
                            ?.EmploymentType
                },

                candidate = new
                {
                    candidateProfileId =
                        application.CandidateProfileId,

                    fullName =
                        application.CandidateProfile
                            ?.User?.FullName,

                    email =
                        application.CandidateProfile
                            ?.User?.Email,

                    phone =
                        application.CandidateProfile
                            ?.Phone,

                    bio =
                        application.CandidateProfile
                            ?.Bio,

                    experienceYears =
                        application.CandidateProfile
                            ?.ExperienceYears,

                    education =
                        application.CandidateProfile
                            ?.Education,

                    skills =
                        application.CandidateProfile
                            ?.Skills,

                    currentJobTitle =
                        application.CandidateProfile
                            ?.CurrentJobTitle,

                    linkedInUrl =
                        application.CandidateProfile
                            ?.LinkedInUrl,

                    portfolioUrl =
                        application.CandidateProfile
                            ?.PortfolioUrl
                }
            });
        }

        private int? GetCurrentUserId()
        {
            string? userIdValue =
                User.FindFirstValue(
                    ClaimTypes.NameIdentifier
                );

            if (int.TryParse(
                    userIdValue,
                    out int userId))
            {
                return userId;
            }

            return null;
        }

        private static decimal CalculateOverallScore(
            CreateEvaluationDto request)
        {
            decimal total =
                request.TechnicalScore +
                request.CommunicationScore +
                request.ExperienceScore +
                request.ProblemSolvingScore +
                request.CultureFitScore;

            return Math.Round(total / 5, 2);
        }
    }
}
